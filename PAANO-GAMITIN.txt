┏━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┓
┃  STEP 1: I-START ANG APP                                      ┃
┗━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┛

   Pinakamadali:
   
   1. Double-click ang file na:  START-HERE.bat
   
   2. Maghintay ng 10 seconds habang nag-start ang 2 terminals
   
   3. Bubuksan ang 2 windows:
      • Backend Terminal (itim na window)
      • Frontend Terminal (itim na window)
   
   4. Kapag nakita mo ang "ready" sa both windows, TAPOS NA!

   Buksan ang Chrome/Edge/Firefox at pumunta sa:
   
   http://localhost:5173
   
   (O kung may ibang port number, gamitin yun - makikita mo sa terminal)

┏━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┓
┃  STEP 4: GAMITIN ANG FEATURES                                 ┃
┗━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┛

   Makikita mo ang 4 main sections:
   
   📊 DASHBOARD
      • Tingnan ang 5 KPIs (New Leads, Active Deals, etc.)
      • Pipeline summary per stage
      • Priority tasks
   
   📁 CUSTOMER DATABASE
      • Mag-add ng bagong customer (lead)
      • I-update ang status (New → Working → Qualified → Converted)
      • Tingnan ang customer details
   
   💼 PIPELINE
      • Mag-add ng bagong deal
      • I-drag ang deals sa different stages (Kanban board)
      • Tingnan ang deal value at probability
   
   ✅ TASKS
      • Mag-add ng follow-up tasks
      • I-mark as completed
      • Filter by status (Open/Completed)

┃  ADMIN FEATURES (Admin account lang)                          ┃
┗━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┛

   Kapag naka-login ka as Admin:
   
   📈 ANALYTICS
      • Branch performance table
      • Role distribution
      • Audit log (recent changes)
   
   👥 ACCOUNT MANAGEMENT
      • Create new users
      • Edit user details
      • Delete users
      • Assign roles and branches
   
   👤 PROFILE
      • Change username
      • Change password

┏━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┓
┃  PAANO MAG-ADD NG CUSTOMER (LEAD)                             ┃
┗━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┛

   1. Click "Customer Database" sa sidebar
   2. Click "New customer" button (upper right)
   3. Fill up ang form:
      • Customer Name (required)
      • Contact Number
      • Address
      • Region
      • Sales Rep (SR)
   4. Click "Save customer"
   5. Makikita mo na ang bagong customer sa list!

┏━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┓
┃  PAANO MAG-ADD NG DEAL                                        ┃
┗━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┛

   1. Click "Pipeline" sa sidebar
   2. Click "New deal" button (upper right)
   3. Fill up ang form:
      • Deal Name (required)
      • Company Name
      • Contact Name
      • Value (amount in PHP)
      • Stage (New Opportunity, Qualified, etc.)
      • Expected Close Date
      • Owner (sales rep)
   4. Click "Save deal"
   5. Makikita mo ang deal sa Kanban board!

┏━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┓
┃  PAANO MAG-ADD NG TASK                                        ┃
┗━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┛

   1. Click "Tasks" sa sidebar
   2. Click "Add task" button (upper right)
   3. Fill up ang form:
      • Task Title (required)
      • Type (Call, Follow-up, Meeting, Email)
      • Owner (sales rep)
      • Deal (optional - link to a deal)
      • Due Date
      • Priority (Low, Medium, High)
      • Notes
   4. Click "Save task"
   5. Makikita mo ang task sa list!

┃  PAANO I-STOP ANG APP                                         ┃
┗━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┛

   2. Press Ctrl+C
   3. Pumunta sa Frontend Terminal (itim na window)

   TAPOS NA! Safe to close lahat.

┃  COMMON PROBLEMS & SOLUTIONS                                  ┃
┗━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┛

   ❌ Problem: "Port already in use"
   ✅ Solution: Close other terminals running the app, then try again
   
   ❌ Problem: "Database connection failed"
   ✅ Solution: Make sure MySQL is running (check XAMPP/WAMP)
   
   ❌ Problem: "Invalid credentials"
   ✅ Solution: Check username/password/branch spelling
   
   ❌ Problem: Blank page / not loading
   ✅ Solution: Wait 10-15 seconds for servers to fully start
   
   ❌ Problem: No data showing
   ✅ Solution: Normal for fresh install - add data via UI
